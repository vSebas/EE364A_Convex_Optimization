\section*{A13.2 True or False (Flop Counts)}

\begin{enumerate}[label=(\alph*)]
    \item Algorithm flop counts allow for very accurate prediction of running time on a given computer}

    \textbf{False}

    \item Since matrix multiplication is associative, the flop count for multiplying three or more matrices doesn't depend on the order in which you multiply them}

    \textbf{False}

    \item Suppose $A \in \mathbf{R}^{n \times n}$ is lower triangular. The flop count for computing $Ab$ is the same order as the flop count for computing $A^{-1}b$}

    \textbf{True}
\end{enumerate}
